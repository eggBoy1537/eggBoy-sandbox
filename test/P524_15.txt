〇convert (整数型 : num, 文字型の配列 : str)
論理型 : minus
整数型 : i, j
文字型の配列 : table
文字型 : tmp
minus = false
i = 0
j = 0
table = "0123456789"

/* 整数値が負であるかどうかチェック */
if (num < 0)
    minus = true
    num = -num
endif

/* 数値の下位から順に文字に変換 */
do
    /* 1桁の数字を格納 */
    str[j] = table[num mod 10]
    j = j + 1
    
    /* 次の桁を取り出す準備 */
    num = num / 10

    /* 格納した数字の数 i をカウントアップ */
    i = i + 1

    /* 3桁ごとに区切りのカンマを挿入 */
    if (i mod 3 = 0 and num ≠ 0)
        str[j] = ","
        j = j + 1
while (num ≠ 0) /* num ≠ 0である限りは18行目のdoに戻って繰り返し処理を実施し数字とカンマを格納していく */


    /* 整数値が負ならマイナス符号をつける */
    if (minus = true)
        str[j] = "-"
        j = j + 1
    endif
    str[j] = EOS
    
    /* strの要素の順序を逆にする */
    i = 0
    j = j - 1
    while (i < j)
        tmp = str[i]
        str[i] = str[j]
        str[j] = tmp
        i = i + 1
        j = j - 1
    endwhile
